関数の定義

functionで関数を宣言する

    function hoge(arg) {
        return arg * arg;
    }
関数の呼び出し
()を付けることによって関数を実行する 
    hoge(10);

無名関数

名前を付けない関数
関数リテラルと呼ばれる式で記述
  • 命名する必要性のない関数をシンプルに記述出来る
  • 関数の引数などに渡せる
    •     var hoge = function(arg) {
        return arg * arg;
    };
    hoge(10);

    var foo = function(func, arg) {
        return func(arg) * arg;
    };
    foo(hoge, 10);
    即時関数
  • 関数を定義し即座に実行出来る
  • グローバルスコープを汚染することなくコードを記述出来る
    •     (function(arg) {
        return arg * arg;
    })(10);

    クロージャ

    コードとスコープが対になったものをクロージャと呼ぶ
  • 関数の中に関数を定義出来る
  • 関数を返す関数を作れる
    •     function outer() {
        var val = 0;

        function inner() {
            val++;
            return val;
        }
        return inner;
    }
    var foo = outer();
    foo();
    foo()を連続して実行してみてください
  • プライベート変数を擬似的に定義出来る(状態を保持)
    • ※グローバルスコープのfooがinner()への参照を保持している

    クラス

    厳密にクラスというものは存在しない
    コンストラクタ
        var d = new Date();

    function Rectangle(w, h) {
        this.width  = w;
        this.height = h;
    }

    var rect = new Rectangle(2, 4);
    プロトタイプ
    面積を計算するメソッドを追加してみる 
        function Rectangle(w, h) {
        this.width  = w;
        this.height = h;
        this.area = function() {
            return this.width * this.height;
        }
    }

    var rect = new Rectangle(2, 4);
    var a = rect.area();
    オブジェクトを生成する度にareaプロパティが再定義されてしまう
    全てのインスタンスで共有するプロパティについては、プロトタイプオブジェクトに追加する 
        function Rectangle(w, h) {
        this.width  = w;
        this.height = h;
    }

    Rectangle.prototype.area = function() {
        return this.width * this.height;
    }
    全ての関数は定義時にprototypeプロパティが生成される
    オブジェクトはプロトタイプからプロパティを継承する
    プライベートメンバ
    クロージャを使って実現出来る
    各メソッドはオブジェクトのインスタンスに保存する、プロトタイプオブジェクトからの継承は不可 
        function Rectangle(w, h) {
        this.getWidth  = function() {return w;}
        this.getHeight = function() {return h;}
    }

    (function (root, factory) {

    root.circle = factory(root.jQuery);

}(this, function ($) {

    function Circle() {
        this.r = 0;
    }

    Circle.PI = 3.14159;

    Circle.prototype.setRadius = function(r) {
        this.r = r;
    }

    Circle.prototype.area = function() {
        return Circle.PI * this.r * this.r;
    }

    return new Circle();
}));

circle.setRadius(10);
